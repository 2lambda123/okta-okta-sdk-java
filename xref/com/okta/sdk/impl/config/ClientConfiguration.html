17  package com.okta.sdk.impl.config;
18  
19  import com.okta.commons.http.config.BaseUrlResolver;
20  import com.okta.commons.http.config.HttpClientConfiguration;
21  import com.okta.commons.lang.Strings;
22  import com.okta.sdk.client.AuthenticationScheme;
23  import com.okta.sdk.client.AuthorizationMode;
24  import com.okta.sdk.impl.api.ClientCredentialsResolver;
25  
26  import java.util.HashSet;
27  import java.util.Set;
28  
29  /**
30   * This class holds the default configuration properties.
31   *
32   * During application initialization all the properties found in the pre-defined locations that are
33   * defined by the user will be added here in the order defined in {@link com.okta.sdk.impl.client.DefaultClientBuilder}.
34   * Unset values will use default values from {@code com/okta/sdk/config/okta.yaml}.
35   *
36   * @since 0.5.0
37   */
38  public class ClientConfiguration extends HttpClientConfiguration {
39  
40      private String apiToken;
41      private ClientCredentialsResolver clientCredentialsResolver;
42      private AuthenticationScheme authenticationScheme;
43      private BaseUrlResolver baseUrlResolver;
44      private AuthorizationMode authorizationMode;
45      private String clientId;
46      private Set<String> scopes = new HashSet<>();
47      private String privateKey;
48      private String kid;
49  
50      public String getApiToken() {
51          return apiToken;
52      }
53  
54      public void setApiToken(String apiToken) {
55          this.apiToken = apiToken;
56      }
57  
58      public ClientCredentialsResolver getClientCredentialsResolver() {
59          return clientCredentialsResolver;
60      }
61  
62      public void setClientCredentialsResolver(ClientCredentialsResolver clientCredentialsResolver) {
63          this.clientCredentialsResolver = clientCredentialsResolver;
64      }
65  
66      public AuthenticationScheme getAuthenticationScheme() {
67          return authenticationScheme;
68      }
69  
70      public void setAuthenticationScheme(AuthenticationScheme authenticationScheme) {
71          this.authenticationScheme = authenticationScheme;
72      }
73  
74      public BaseUrlResolver getBaseUrlResolver() {
75          return baseUrlResolver;
76      }
77  
78      public void setBaseUrlResolver(BaseUrlResolver baseUrlResolver) {
79          this.baseUrlResolver = baseUrlResolver;
80      }
81  
82      public AuthorizationMode getAuthorizationMode() {
83          return authorizationMode;
84      }
85  
86      public void setAuthorizationMode(AuthorizationMode authorizationMode) {
87          this.authorizationMode = authorizationMode;
88      }
89  
90      public String getClientId() {
91          return clientId;
92      }
93  
94      public void setClientId(String clientId) {
95          this.clientId = clientId;
96      }
97  
98      public Set<String> getScopes() {
99          return scopes;
100     }
101 
102     public void setScopes(Set<String> scopes) {
103         this.scopes = scopes;
104     }
105 
106     public String getPrivateKey() {
107         return privateKey;
108     }
109 
110     public void setPrivateKey(String privateKey) {
111         this.privateKey = privateKey;
112     }
113 
114     public String getKid() {
115         return kid;
116     }
117 
118     public void setKid(String kid) {
119         this.kid = kid;
120     }
121 
122     @Override
123     public String getBaseUrl() {
124         String baseUrl = super.getBaseUrl();
125 
126         if (Strings.isEmpty(baseUrl) && baseUrlResolver != null) {
127             baseUrl = baseUrlResolver.getBaseUrl();
128         }
129 
130         return baseUrl;
131     }
132 
133     @Override
134     public String toString() {
135         return "ClientConfiguration { baseUrl='" + getBaseUrl() + '\'' +
136             ", authorizationMode=" + getAuthorizationMode() +
137             ", clientId=" + getClientId() +
138             ", scopes=" + getScopes() +
139             ", privateKey=" + ((getPrivateKey() != null) ? "*****" : null) +
140             ", connectionTimeout=" + getConnectionTimeout() +
141             ", requestAuthenticator=" + getRequestAuthenticator() +
142             ", retryMaxElapsed=" + getRetryMaxElapsed() +
143             ", retryMaxAttempts=" + getRetryMaxAttempts() +
144             ", proxy=" + getProxy() +
145             " }";
146     }
147 }
