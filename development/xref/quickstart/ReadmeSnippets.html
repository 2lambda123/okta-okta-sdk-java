240     private void callAnotherEndpoint() {
241 
242         ApiClient apiClient = buildApiClient("orgBaseUrl", "apiKey");
243 
244         // Create a BookmarkApplication
245         BookmarkApplication bookmarkApplication = new BookmarkApplication();
246         bookmarkApplication.setName("bookmark");

248         bookmarkApplication.setSignOnMode(ApplicationSignOnMode.BOOKMARK);

255         bookmarkApplication.setSettings(bookmarkApplicationSettings);
256 
257         ResponseEntity<BookmarkApplication> responseEntity = apiClient.invokeAPI("/api/v1/apps",
258             HttpMethod.POST,
259             Collections.emptyMap(),
260             null,
261             bookmarkApplication,
262             new HttpHeaders(),
263             new LinkedMultiValueMap<>(),
264             null,
265             Collections.singletonList(MediaType.APPLICATION_JSON),
266             MediaType.APPLICATION_JSON,
267             new String[]{"API Token"},
268             new ParameterizedTypeReference<BookmarkApplication>() {});
269 
270         BookmarkApplication createdApp = responseEntity.getBody();
271     }

273     private void paging() {
274 
275         UserApi userApi = new UserApi(client);
276 
277         // limit
278         int pageSize = 2;
279 
280         PagedList<User> usersPagedListOne = userApi.listUsersWithPaginationInfo(null, null, pageSize, null, null, null, null);
281 
282         // e.g. https://example.okta.com/api/v1/users?after=000u3pfv9v4SQXvpBB0g7&limit=2
283         String nextPageUrl = usersPagedListOne.getNextPage();
284 
285         // replace 'after' with actual cursor from the nextPageUrl
286         PagedList<User> usersPagedListTwo = userApi.listUsersWithPaginationInfo("after", null, pageSize, null, null, null, null);
287 
288         // loop through all of them (paging is automatic)
289         for (User tmpUser : usersPagedListOne.getItems()) {
290             log.info("User: {}", tmpUser.getProfile().getEmail());
291         }
292 
293         // or stream
294         usersPagedListOne.getItems().forEach(tmpUser -> log.info("User: {}", tmpUser.getProfile().getEmail()));
295     }

297     private void complexCaching() {
298         Caches.newCacheManager()
299             .withDefaultTimeToLive(300, TimeUnit.SECONDS) // default
300             .withDefaultTimeToIdle(300, TimeUnit.SECONDS) //general default
301             .withCache(forResource(User.class) //User-specific cache settings
302                 .withTimeToLive(1, TimeUnit.HOURS)
303                 .withTimeToIdle(30, TimeUnit.MINUTES))
304             .withCache(forResource(Group.class) //Group-specific cache settings
305                 .withTimeToLive(1, TimeUnit.HOURS))
306             //... etc ...
307             .build(); //build the CacheManager
308     }
