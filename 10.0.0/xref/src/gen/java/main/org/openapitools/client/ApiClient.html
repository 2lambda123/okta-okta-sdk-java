725                                         }
726                                         } else {
727                                         // return data from cache
728                                         return cache.get(cacheKey);
729                                         }
730                                         } else {
731                                         ResponseEntity<T> responseEntity = retryTemplate.execute(context -> restTemplate.exchange(requestEntity, returnType));
732 
733                                             if (responseEntity != null && responseEntity.getStatusCode().is2xxSuccessful()) {
734                                             if (cache.get(cacheKey) != null) {
735                                             cache.put(cacheKey, responseEntity);
736                                             }
737                                             return responseEntity;
738                                             } else {
739                                             // The error handler built into the RestTemplate should handle 400 and 500 series errors.
740                                             throw new RestClientException("API returned " + responseEntity.getStatusCode() + " and it wasn't handled by the RestTemplate error handler");
741                                             }
742                                             }
743                                     }

747                                     * @param headers The headers to add
748                                     * @param requestBuilder The current request

751                                     for (Entry<String, List<String>> entry : headers.entrySet()) {
752                                         List<String> values = entry.getValue();
753                                             for (String value : values) {

763                                             *
764                                             * @param cookies        The cookies to add
765                                             * @param requestBuilder The current request
766                                             */
767                                             protected void addCookiesToRequest(MultiValueMap<String, String> cookies, BodyBuilder requestBuilder) {

773                                             /**
774                                             * Build cookie header. Keeps a single value per cookie (as per <a href="https://tools.ietf.org/html/rfc6265#section-5.3">
775                                                 * RFC6265 section 5.3</a>).
776                                             *
777                                             * @param cookies map all cookies
778                                             * @return header string for cookies.
779                                             */
780                                             private String buildCookieHeader(MultiValueMap<String, String> cookies) {
781                                             final StringBuilder cookieValue = new StringBuilder();
782                                             String delimiter = "";
783                                             for (final Map.Entry<String, List<String>> entry : cookies.entrySet()) {
784                                                 final String value = entry.getValue().get(entry.getValue().size() - 1);
785                                                 cookieValue.append(String.format("%s%s=%s", delimiter, entry.getKey(), value));
786                                                 delimiter = "; ";
787                                                 }
788                                                 return cookieValue.toString();
789                                                 }

799

867                                                 log.info("HTTP Status Code: " + response.getRawStatusCode());
868                                                 log.info("Status Text: " + response.getStatusText());
869                                                 log.info("HTTP Headers: " + headersToString(response.getHeaders()));
870                                                 log.info("Response Body: " + bodyToString(response.getBody()));
871                                                 }

873                                                 private String headersToString(HttpHeaders headers) {
874                                                 if(headers == null || headers.isEmpty()) {
875                                                 return "";
876                                                 }
877                                                 StringBuilder builder = new StringBuilder();
878                                                 for (Entry<String, List<String>> entry : headers.entrySet()) {
879                                                     builder.append(entry.getKey()).append("=[");
880                                                     for (String value : entry.getValue()) {
881                                                     if (entry.getKey().equals("Authorization")) {
882                                                     builder.append("*****").append(",");
883                                                     } else {
884                                                     builder.append(value).append(",");
885                                                     }
886                                                     }
887                                                     builder.setLength(builder.length() - 1); // Get rid of trailing comma
888                                                     builder.append("],");
889                                                     }
890                                                     builder.setLength(builder.length() - 1); // Get rid of trailing comma
891                                                     return builder.toString();
892                                                     }

894                                                     private String bodyToString(InputStream body) throws IOException {
895                                                     StringBuilder builder = new StringBuilder();
896                                                     BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(body, StandardCharsets.UTF_8));
897                                                     String line = bufferedReader.readLine();
898                                                     while (line != null) {
899                                                     builder.append(line).append(System.lineSeparator());
900                                                     line = bufferedReader.readLine();
901                                                     }
902                                                     bufferedReader.close();
903                                                     return builder.toString();
904                                                     }
